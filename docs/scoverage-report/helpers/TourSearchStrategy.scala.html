44   def getNextStep(previousChunk: Position, previousRequest: TourChunkRequest): Step = {
45 
46     // Get current position relative to the full board
47     val actualCoords = previousRequest.chunkID.toActual(previousChunk.square.asTuple)
48     val currentSquare = fullBoard.getSquare(actualCoords).get
49 
50     // Get the visited squares from all the previous chunks
51     val totalVisited = previousRequest.chunkIDs.flatMap(_.toActual(previousChunk.visited))
52 
53     // Get best possible escape direction, using the standard algorithm as the rest of the search
54     previousChunk.copy(square = currentSquare, visited = totalVisited).nextSteps.head
55   }

57   def getNextTourChunkRequest(previousChunk: Position, previousRequest: TourChunkRequest): Option[TourChunkRequest] = {
58 
59     val nextStep = getNextStep(previousChunk, previousRequest)

61   }

63   def assembleChunks(lastChunkRequest: TourChunkRequest, first: Position, second: Position, third: Position, fourth: Position): Option[Position] = {
64     val lastSquare: Square = fullBoard.getSquare(lastChunkRequest.chunkID.toActual(fourth.square.asTuple)).get
65     Some(Position(lastSquare, fourth.history ::: third.history ::: second.history ::: first.history, Nil))
66   }
